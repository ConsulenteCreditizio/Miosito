        // Mostra/Nascondi Chatbot
        const chatbotBtn = document.getElementById("chatbot-btn");
        const chatbotPanel = document.getElementById("chatbot-panel");
        chatbotBtn.addEventListener("click", () => {
            chatbotPanel.style.display = chatbotPanel.style.display === "flex" ? "none" : "flex";
        });

        // Funzione invio messaggio
        async function sendMessage() {
            const input = document.getElementById("userInput");
            const message = input.value.trim();
            if (!message) return;

            const messagesDiv = document.getElementById("chatbot-messages");
            messagesDiv.innerHTML += `<div class="message user"><strong>Tu:</strong> ${message}</div>`;
            input.value = "";
            messagesDiv.scrollTop = messagesDiv.scrollHeight;

            try {
                const response = await fetch("http://localhost:3000/chat", {
                    method: "POST",
                    headers: { "Content-Type": "application/json" },
                    body: JSON.stringify({ message })
                });
                const data = await response.json();
                messagesDiv.innerHTML += `<div class="message ai"><strong>Massimo Macis:</strong> ${data.reply}</div>`;
                messagesDiv.scrollTop = messagesDiv.scrollHeight;
            } catch (error) {
                messagesDiv.innerHTML += `<div class="message ai">Errore nella connessione al server.</div>`;
            }
        }
    </script>
